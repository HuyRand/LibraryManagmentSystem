create table MEMBER
( MEMID  int auto_increment primary key ,
  NAME VARCHAR(50) NOT NULL,
  DOB DATETIME,
  NUMBER_OF_BOOK_ALLOWED INTEGER(5),
  EMAIL VARCHAR(50),
  ADDRESS VARCHAR (80),
  PENALTY INT(20),
  CLASS VARCHAR(50),
  MAJOR VARCHAR(50)
);


CREATE TABLE BOOK (
  BOOKID int  auto_increment primary key,
  NAME varchar(50),
  PUBLISHER varchar(50),
  AUTHOR Varchar(50),
  CATEGORY varchar(20),
  AVAILABLE bool,
  PRICE int(20),
  YEAR int(10),
  BOOK_IMG_ID int (5)
);

CREATE TABLE BORROW (
  MEMID INTEGER NOT NULL REFERENCES MEMBER(MEMID),
  BOOKID INTEGER NOT NULL REFERENCES BOOK(BOOKID),
  RESERVE_DATE date,
  DUE_DATE date,
  RETURN_DATE date,
  CONSTRAINT PK_BORROW PRIMARY KEY (MEMID,BOOKID)
  );

CREATE TABLE LOCATION (
  LOCA_ID int auto_increment primary key,
  BOOKID INTEGER REFERENCES BOOK(BOOKID),
  SHELF_ID intEGER(5),
  SECTION_ID intEGER(5), 
  ROW_NUMBER intEGER(5)
  );
CREATE TABLE BORROW_LOG(
  BORROWID int  auto_increment primary key,
  MEMID INTEGER NOT NULL REFERENCES MEMBER(MEMID),
  BOOKID INTEGER NOT NULL REFERENCES BOOK(BOOKID),
  RESERVE_DATE date,
  DUE_DATE date,
  RETURN_DATE date,
  AUTHOR VARCHAR(50),
  CATEGORY VARCHAR(50)
);
