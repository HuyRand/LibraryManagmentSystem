create table MEMBER
( MEMID  int auto_increment primary key ,
  NAME VARCHAR(50) NOT NULL,
  DOB DATETIME,
  NUMBER_OF_BOOK_ALLOWED INTEGER(5),
  EMAIL VARCHAR(50),
  ADDRESS VARCHAR (80),
  BILL INT
);


CREATE TABLE BOOK (
  BOOKID int  auto_increment primary key,
  PUBLISHER varchar(50),
  AUTHOR Varchar(50),
  CATEGORY varchar(20),
  AVAILABLE bool,
  PRICE int(20),
  year int(10),
  BOOK_IMG_ID int (5)
);

CREATE TABLE BORROW (
  MEMID INTEGER NOT NULL REFERENCES MEMBER(MEMID),
  BOOKID INTEGER NOT NULL REFERENCES BOOK(BOOKID),
  RESERVE_DATE datetime,
  DUE_DATE datetime,
  RETURN_DATE datetime,
  CONSTRAINT PK_BORROW PRIMARY KEY (MEMID,BOOKID)
  );

CREATE TABLE LOCATION (
  LOCA_ID varchar (5) primary key,
  BOOKID INTEGER REFERENCES BOOK(BOOKID),
  SHELF_ID intEGER(5),
  SECTION_ID intEGER(5), 
  ROW_NUMBER intEGER(5)
  );
